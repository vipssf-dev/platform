import { sql } from "drizzle-orm";
import { pgTable, text, varchar, boolean, integer, jsonb, timestamp, serial } from "drizzle-orm/pg-core";
import { createInsertSchema } from "drizzle-zod";
import { z } from "zod";

export const users = pgTable("users", {
  id: varchar("id").primaryKey().default(sql`gen_random_uuid()`),
  username: text("username").notNull().unique(),
  password: text("password").notNull(),
});

export const insertUserSchema = createInsertSchema(users).pick({
  username: true,
  password: true,
});

export type InsertUser = z.infer<typeof insertUserSchema>;
export type User = typeof users.$inferSelect;

export const attachmentSchema = z.object({
  name: z.string(),
  data: z.string(),
  type: z.string(),
});

export type AttachmentData = z.infer<typeof attachmentSchema>;

export const actionSchema = z.object({
  text: z.string(),
  date: z.string(),
  period: z.string(),
  attachments: z.array(attachmentSchema).optional(),
});

export type ActionData = z.infer<typeof actionSchema>;

export const referrals = pgTable("referrals", {
  id: varchar("id").primaryKey().default(sql`gen_random_uuid()`),
  studentId: text("student_id").notNull(),
  studentName: text("student_name").notNull(),
  grade: text("grade").notNull(),
  section: text("section").notNull(),
  counselor: text("counselor").notNull(),
  caseStudy: boolean("case_study"),
  caseStudyFile: jsonb("case_study_file").$type<AttachmentData | null>(),
  violationDegree: integer("violation_degree"),
  violationText: text("violation_text"),
  vicePrincipal: text("vice_principal"),
  vicePrincipalDate: text("vice_principal_date"),
  actionTaken: text("action_taken"),
  actions: jsonb("actions").$type<ActionData[]>().default([]),
  status: text("status").default("sent"),
  createdAt: timestamp("created_at").defaultNow(),
});

export const insertReferralSchema = createInsertSchema(referrals).omit({
  id: true,
  createdAt: true,
});

export type InsertReferral = z.infer<typeof insertReferralSchema>;
export type Referral = typeof referrals.$inferSelect;

export const evidenceFileSchema = z.object({
  name: z.string(),
  data: z.string(),
  type: z.string(),
});

export type EvidenceFile = z.infer<typeof evidenceFileSchema>;

export const guidancePlanEntries = pgTable("guidance_plan_entries", {
  id: varchar("id").primaryKey(),
  programName: text("program_name").notNull(),
  objectives: text("objectives").notNull(),
  startDate: text("start_date"),
  endDate: text("end_date"),
  responsibility: text("responsibility"),
  executionStatus: text("execution_status"),
  evidenceFiles: jsonb("evidence_files").$type<EvidenceFile[]>().default([]),
  updatedAt: timestamp("updated_at").defaultNow(),
});

export const insertGuidancePlanEntrySchema = createInsertSchema(guidancePlanEntries).omit({
  updatedAt: true,
});

export type InsertGuidancePlanEntry = z.infer<typeof insertGuidancePlanEntrySchema>;
export type GuidancePlanEntry = typeof guidancePlanEntries.$inferSelect;

export const sessionSchema = z.object({
  date: z.string(),
  content: z.string(),
  recommendations: z.string(),
});

export type SessionData = z.infer<typeof sessionSchema>;
