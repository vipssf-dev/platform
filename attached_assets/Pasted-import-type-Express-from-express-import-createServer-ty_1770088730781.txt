import type { Express } from "express";
import { createServer, type Server } from "http";
import { storage } from "./storage";
import { insertActivitySchema } from "@shared/schema";
import { z } from "zod";

export async function registerRoutes(
  httpServer: Server,
  app: Express
): Promise<Server> {

  // Get all activities
  app.get("/api/activities", async (req, res) => {
    try {
      const activities = await storage.getAllActivities();
      res.json(activities);
    } catch (error) {
      console.error("Fetch activities error:", error);
      res.status(500).json({ error: "Failed to fetch activities" });
    }
  });

  // Get single activity
  app.get("/api/activities/:id", async (req, res) => {
    try {
      const activity = await storage.getActivity(req.params.id);
      if (!activity) {
        return res.status(404).json({ error: "Activity not found" });
      }
      res.json(activity);
    } catch (error) {
      console.error("Fetch activity error:", error);
      res.status(500).json({ error: "Failed to fetch activity" });
    }
  });

  // Create new activity
  app.post("/api/activities", async (req, res) => {
    try {
      console.log("Received POST /api/activities:", JSON.stringify(req.body));
      
      // Clean up empty strings to null for optional fields
      const cleanedData = {
        semester: req.body.semester || null,
        activity: req.body.activity || null,
        field: req.body.field || null,
        executionStartDate: req.body.executionStartDate || null,
        executionEndDate: req.body.executionEndDate || null,
        executionLevel: req.body.executionLevel || null,
        executionResponsibility: req.body.executionResponsibility || null,
        participantsCount: req.body.participantsCount ?? null,
        executionStatus: req.body.executionStatus || null,
        achievedPosition: req.body.achievedPosition || null,
      };
      
      console.log("Cleaned data:", JSON.stringify(cleanedData));
      
      const validatedData = insertActivitySchema.parse(cleanedData);
      console.log("Validated data:", JSON.stringify(validatedData));
      
      const activity = await storage.createActivity(validatedData);
      console.log("Created activity:", JSON.stringify(activity));
      
      res.status(201).json(activity);
    } catch (error) {
      console.error("Create activity error:", error);
      if (error instanceof z.ZodError) {
        console.error("Zod validation errors:", JSON.stringify(error.errors));
        return res.status(400).json({ error: "Invalid data", details: error.errors });
      }
      const errorMessage = error instanceof Error ? error.message : "Unknown error";
      console.error("Error message:", errorMessage);
      res.status(500).json({ error: "Failed to create activity", details: errorMessage });
    }
  });

  // Update activity
  app.patch("/api/activities/:id", async (req, res) => {
    try {
      // Clean up empty strings to null for optional fields
      const cleanedData: Record<string, unknown> = {};
      const allowedFields = [
        'semester', 'activity', 'field', 'executionStartDate', 'executionEndDate',
        'executionLevel', 'executionResponsibility', 'participantsCount', 
        'executionStatus', 'achievedPosition'
      ];
      
      for (const field of allowedFields) {
        if (req.body[field] !== undefined) {
          if (field === 'participantsCount') {
            cleanedData[field] = req.body[field] ?? null;
          } else {
            cleanedData[field] = req.body[field] || null;
          }
        }
      }
      
      const validatedData = insertActivitySchema.partial().parse(cleanedData);
      const activity = await storage.updateActivity(req.params.id, validatedData);
      if (!activity) {
        return res.status(404).json({ error: "Activity not found" });
      }
      res.json(activity);
    } catch (error) {
      console.error("Update activity error:", error);
      if (error instanceof z.ZodError) {
        return res.status(400).json({ error: "Invalid data", details: error.errors });
      }
      res.status(500).json({ error: "Failed to update activity" });
    }
  });

  // Delete activity
  app.delete("/api/activities/:id", async (req, res) => {
    try {
      const deleted = await storage.deleteActivity(req.params.id);
      if (!deleted) {
        return res.status(404).json({ error: "Activity not found" });
      }
      res.status(204).send();
    } catch (error) {
      console.error("Delete activity error:", error);
      res.status(500).json({ error: "Failed to delete activity" });
    }
  });

  return httpServer;
}
