import { pgTable, text, varchar, integer, date, time, boolean } from "drizzle-orm/pg-core";
import { createInsertSchema } from "drizzle-zod";
import { z } from "zod";

export const grades = pgTable("grades", {
  id: varchar("id").primaryKey(),
  name: text("name").notNull(),
});

export const classrooms = pgTable("classrooms", {
  id: varchar("id").primaryKey(),
  name: text("name").notNull(),
  gradeId: varchar("grade_id").notNull(),
});

export const students = pgTable("students", {
  id: varchar("id").primaryKey(),
  name: text("name").notNull(),
  gradeId: varchar("grade_id").notNull(),
  classroomId: varchar("classroom_id").notNull(),
});

export const medicalCases = pgTable("medical_cases", {
  id: varchar("id").primaryKey(),
  studentName: text("student_name").notNull(),
  grade: text("grade").notNull(),
  classroomSection: text("classroom_section").notNull(),
  disease: text("disease").notNull(),
  hasMedicalReport: boolean("has_medical_report").notNull().default(false),
});

export const dailyTracking = pgTable("daily_tracking", {
  id: varchar("id").primaryKey(),
  studentId: varchar("student_id").notNull(),
  date: text("date").notNull(),
  caseType: text("case_type").notNull(),
  action: text("action").notNull(),
  notes: text("notes"),
});

export const evacuationDrills = pgTable("evacuation_drills", {
  id: varchar("id").primaryKey(),
  drillNumber: integer("drill_number").notNull(),
  day: text("day").notNull(),
  date: text("date").notNull(),
  startTime: text("start_time").notNull(),
  endTime: text("end_time").notNull(),
  duration: integer("duration").notNull(),
  location: text("location").notNull(),
  safeZoneExists: boolean("safe_zone_exists").notNull(),
  evacuationType: text("evacuation_type").notNull(),
  planAvailable: boolean("plan_available").notNull(),
  evacuationSpeed: text("evacuation_speed").notNull(),
  emergencySignsQuality: text("emergency_signs_quality").notNull(),
  alarmStatus: text("alarm_status").notNull(),
  alarmNotes: text("alarm_notes"),
  guideResponse: text("guide_response").notNull(),
  guidePosition: text("guide_position").notNull(),
  guidePerformance: text("guide_performance").notNull(),
  guideNotes: text("guide_notes"),
  obstacles: text("obstacles").notNull().default("[]"),
  recommendations: text("recommendations"),
});

export const supervisionPlan = pgTable("supervision_plan", {
  id: varchar("id").primaryKey(),
  month: text("month"),
  week: integer("week"),
  activity: text("activity").notNull(),
  date: text("date"),
  duration: text("duration"),
  targetAudience: text("target_audience"),
  responsibility: text("responsibility"),
  executionStatus: text("execution_status").notNull().default("لم يتم التنفيذ"),
  evidence: text("evidence").array().default([]),
  completed: boolean("completed").notNull().default(false),
  notes: text("notes"),
});

export const insertGradeSchema = createInsertSchema(grades).omit({ id: true });
export const insertClassroomSchema = createInsertSchema(classrooms).omit({ id: true });
export const insertStudentSchema = createInsertSchema(students).omit({ id: true });
export const insertMedicalCaseSchema = createInsertSchema(medicalCases).omit({ id: true });
export const insertDailyTrackingSchema = createInsertSchema(dailyTracking).omit({ id: true });
export const insertEvacuationDrillSchema = createInsertSchema(evacuationDrills).omit({ id: true });
export const insertSupervisionPlanSchema = createInsertSchema(supervisionPlan).omit({ id: true });

export type InsertGrade = z.infer<typeof insertGradeSchema>;
export type Grade = typeof grades.$inferSelect;

export type InsertClassroom = z.infer<typeof insertClassroomSchema>;
export type Classroom = typeof classrooms.$inferSelect;

export type InsertStudent = z.infer<typeof insertStudentSchema>;
export type Student = typeof students.$inferSelect;

export type InsertMedicalCase = z.infer<typeof insertMedicalCaseSchema>;
export type MedicalCase = typeof medicalCases.$inferSelect;

export type InsertDailyTracking = z.infer<typeof insertDailyTrackingSchema>;
export type DailyTracking = typeof dailyTracking.$inferSelect;

export type InsertEvacuationDrill = z.infer<typeof insertEvacuationDrillSchema>;
export type EvacuationDrill = typeof evacuationDrills.$inferSelect;

export type InsertSupervisionPlan = z.infer<typeof insertSupervisionPlanSchema>;
export type SupervisionPlan = typeof supervisionPlan.$inferSelect;

export const environmentInspectionSchedule = pgTable("environment_inspection_schedule", {
  id: varchar("id").primaryKey(),
  academicYear: text("academic_year").notNull().default("1447"),
  semester: text("semester").notNull(),
  facilityName: text("facility_name").notNull(),
  firstVisitDate: text("first_visit_date"),
  secondVisitDate: text("second_visit_date"),
  notes: text("notes"),
});

export const environmentInspection = pgTable("environment_inspection", {
  id: varchar("id").primaryKey(),
  academicYear: text("academic_year").notNull().default("1447"),
  date: text("date").notNull(),
  category: text("category").notNull(),
  item: text("item").notNull(),
  status: boolean("status"),
  notes: text("notes"),
});

export const unhealthyBehaviors = pgTable("unhealthy_behaviors", {
  id: varchar("id").primaryKey(),
  academicYear: text("academic_year").notNull().default("1447"),
  semester: text("semester").notNull(),
  date: text("date").notNull(),
  studentId: varchar("student_id").notNull(),
  behavior: text("behavior").notNull(),
  detectionMethod: text("detection_method"),
  followUp: text("follow_up"),
});

export const chronicDiseases = pgTable("chronic_diseases", {
  id: varchar("id").primaryKey(),
  academicYear: text("academic_year").notNull().default("1447"),
  semester: text("semester").notNull(),
  date: text("date").notNull(),
  studentId: varchar("student_id").notNull(),
  disease: text("disease").notNull(),
  detectionMethod: text("detection_method"),
  treatment: text("treatment"),
  hasAttachedReport: boolean("has_attached_report").default(false),
});

export const infectiousDiseases = pgTable("infectious_diseases", {
  id: varchar("id").primaryKey(),
  academicYear: text("academic_year").notNull().default("1447"),
  semester: text("semester").notNull(),
  date: text("date").notNull(),
  studentId: varchar("student_id").notNull(),
  condition: text("condition").notNull(),
  referral: text("referral"),
  spreadLocation: text("spread_location"),
});

export const users = pgTable("users", {
  id: varchar("id").primaryKey(),
  username: text("username").notNull().unique(),
  password: text("password").notNull(),
});

export const insertEnvironmentInspectionScheduleSchema = createInsertSchema(environmentInspectionSchedule).omit({ id: true });
export const insertEnvironmentInspectionSchema = createInsertSchema(environmentInspection).omit({ id: true });
export const insertUnhealthyBehaviorsSchema = createInsertSchema(unhealthyBehaviors).omit({ id: true });
export const insertChronicDiseasesSchema = createInsertSchema(chronicDiseases).omit({ id: true });
export const insertInfectiousDiseasesSchema = createInsertSchema(infectiousDiseases).omit({ id: true });

export type InsertEnvironmentInspectionSchedule = z.infer<typeof insertEnvironmentInspectionScheduleSchema>;
export type EnvironmentInspectionSchedule = typeof environmentInspectionSchedule.$inferSelect;

export type InsertEnvironmentInspection = z.infer<typeof insertEnvironmentInspectionSchema>;
export type EnvironmentInspection = typeof environmentInspection.$inferSelect;

export type InsertUnhealthyBehaviors = z.infer<typeof insertUnhealthyBehaviorsSchema>;
export type UnhealthyBehaviors = typeof unhealthyBehaviors.$inferSelect;

export type InsertChronicDiseases = z.infer<typeof insertChronicDiseasesSchema>;
export type ChronicDiseases = typeof chronicDiseases.$inferSelect;

export type InsertInfectiousDiseases = z.infer<typeof insertInfectiousDiseasesSchema>;
export type InfectiousDiseases = typeof infectiousDiseases.$inferSelect;

export const insertUserSchema = createInsertSchema(users).pick({
  username: true,
  password: true,
});

export type InsertUser = z.infer<typeof insertUserSchema>;
export type User = typeof users.$inferSelect;
