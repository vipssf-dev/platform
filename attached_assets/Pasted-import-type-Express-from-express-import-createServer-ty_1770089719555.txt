  // Fix missing classrooms - adds any classrooms that don't exist
  app.post("/api/classrooms/fix-missing", async (req, res) => {
    try {
      const { db } = await import("./db");
      const { classrooms, grades } = await import("@shared/schema");
      const { eq } = await import("drizzle-orm");
      
      const allGrades = await storage.getGrades();
      const allClassrooms = await storage.getClassrooms();
      
      // Grade 6 has only 3 classrooms, others have 4
      const classroomCounts: Record<string, number> = {
        "grade-1": 4, "grade-2": 4, "grade-3": 4,
        "grade-4": 4, "grade-5": 4, "grade-6": 3,
      };
      
      const gradeNames: Record<string, string> = {
        "grade-1": "الصف الأول", "grade-2": "الصف الثاني", "grade-3": "الصف الثالث",
        "grade-4": "الصف الرابع", "grade-5": "الصف الخامس", "grade-6": "الصف السادس",
      };
      
      let addedCount = 0;
      
      for (const gradeId of Object.keys(classroomCounts)) {
        const numClassrooms = classroomCounts[gradeId];
        const gradeName = gradeNames[gradeId];
        const gradeNumber = gradeId.replace("grade-", "");
        
        for (let i = 1; i <= numClassrooms; i++) {
          const classroomId = `classroom-${gradeNumber}-${i}`;
          const classroomName = `${gradeName} / ${i}`;
          
          // Check if classroom exists
          const exists = allClassrooms.find(c => c.id === classroomId);
          if (!exists) {
            await db.insert(classrooms).values({
              id: classroomId,
              name: classroomName,
              gradeId: gradeId,
            }).onConflictDoNothing();
            addedCount++;
          }
        }
      }
      
      res.json({ 
        message: addedCount > 0 
          ? `تم إضافة ${addedCount} فصل ناقص` 
          : "جميع الفصول موجودة",
        added: addedCount 
      });
    } catch (error) {
      console.error("Error fixing classrooms:", error);
      res.status(500).json({ error: "حدث خطأ" });
    }
  });

  app.post("/api/classrooms/fix-names", async (req, res) => {
    try {
      const allClassrooms = await storage.getClassrooms();
      const grades = await storage.getGrades();
      
      const letterToNumber: Record<string, string> = {
        "أ": "1", "ب": "2", "ج": "3", "د": "4",
        "A": "1", "B": "2", "C": "3", "D": "4",
      };
      
      let updatedCount = 0;
      
      for (const classroom of allClassrooms) {
        let newName = classroom.name;
        let needsUpdate = false;
        
        for (const [letter, number] of Object.entries(letterToNumber)) {
          if (classroom.name.includes(`/ ${letter}`) || classroom.name.endsWith(` ${letter}`)) {
            const grade = grades.find(g => g.id === classroom.gradeId);
            if (grade) {
              newName = `${grade.name} / ${number}`;
              needsUpdate = true;
              break;
            }
          }
        }
        
        if (needsUpdate) {
          const { db } = await import("./db");
          const { classrooms } = await import("@shared/schema");
          const { eq } = await import("drizzle-orm");
          await db.update(classrooms).set({ name: newName }).where(eq(classrooms.id, classroom.id));
          updatedCount++;
        }
      }
      
      res.json({ 
        updated: updatedCount, 
        message: `تم تحديث ${updatedCount} فصل` 
      });
    } catch (error) {
      console.error("Fix classroom names error:", error);
      res.status(500).json({ error: "فشل في تحديث أسماء الفصول" });
    }
  });

        name: s.name,
        gradeId: s.gradeId,
        classroomId: s.classroomId,
      }));
