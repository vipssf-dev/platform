import type { Express, Request } from "express";
import { createServer, type Server } from "http";
import { storage } from "./storage";
import multer, { type FileFilterCallback } from "multer";
import path from "path";
import fs from "fs";
import * as XLSX from "xlsx";
import { registerObjectStorageRoutes } from "./replit_integrations/object_storage";
import { generateDFDPdfArabic, generateDFDPdfEnglish } from "./pdf-generator";

const excelUpload = multer({
  storage: multer.memoryStorage(),
  limits: { fileSize: 10 * 1024 * 1024 },
  fileFilter: (_req, file, cb) => {
    if (file.mimetype === 'application/vnd.ms-excel' || 
        file.mimetype === 'application/vnd.openxmlformats-officedocument.spreadsheetml.sheet' ||
        file.originalname.endsWith('.xls') ||
        file.originalname.endsWith('.xlsx')) {
      cb(null, true);
    } else {
      cb(new Error('Only Excel files are allowed'));
    }
  }
});

export async function registerRoutes(
  httpServer: Server,
  app: Express
): Promise<Server> {
  
  registerObjectStorageRoutes(app);
  
  app.get("/api/students", async (req, res) => {
    try {
      const students = await storage.getStudents();
      res.json(students);
    } catch (error) {
      res.status(500).json({ error: "Failed to fetch students" });
    }
  });

  app.post("/api/students", async (req, res) => {
    try {
      const student = await storage.createStudent(req.body);
      res.status(201).json(student);
    } catch (error) {
      res.status(500).json({ error: "Failed to create student" });
    }
  });

  app.post("/api/students/import", excelUpload.single("file"), async (req, res) => {
    try {
      if (!req.file) {
        return res.status(400).json({ error: "No file uploaded" });
      }

      const workbook = XLSX.read(req.file.buffer, { type: "buffer" });
      const sheetName = workbook.SheetNames.find(name => name !== 'Sheet1' && name !== 'Sheet3') || workbook.SheetNames[1] || workbook.SheetNames[0];
      const sheet = workbook.Sheets[sheetName];
      const data = XLSX.utils.sheet_to_json(sheet, { header: 1 }) as any[][];

      const gradeMap: Record<string, string> = {
        'الأول الابتدائي': 'أول',
        'الثاني الابتدائي': 'ثاني',
        'الثالث الابتدائي': 'ثالث',
        'الرابع الابتدائي': 'رابع',
        'الخامس الابتدائي': 'خامس',
        'السادس الابتدائي': 'سادس'
      };

      const students = data.slice(1).filter(row => row.length >= 3);
      let imported = 0;
      let errors: string[] = [];

      for (const row of students) {
        const section = String(row[0]);
        const gradeRaw = String(row[1]);
        const name = String(row[2]);
        const grade = gradeMap[gradeRaw] || gradeRaw;

        try {
          await storage.createStudent({ name, grade, section });
          imported++;
        } catch (err: any) {
          errors.push(`فشل استيراد: ${name}`);
        }
      }

      res.json({ 
        success: true, 
        imported, 
        total: students.length,
        errors: errors.slice(0, 10)
      });
    } catch (error: any) {
      console.error("Import error:", error);
      res.status(500).json({ error: "فشل في استيراد الملف: " + error.message });
    }
  });

  app.delete("/api/students", async (req, res) => {
    try {
      await storage.deleteAllStudents();
      res.json({ success: true });
    } catch (error) {
      res.status(500).json({ error: "Failed to delete students" });
    }
  });

  app.get("/api/programs", async (req, res) => {
    try {
      const programs = await storage.getPrograms();
      res.json(programs);
    } catch (error) {
      res.status(500).json({ error: "Failed to fetch programs" });
    }
  });

  app.post("/api/programs", async (req, res) => {
    try {
      const program = await storage.createProgram(req.body);
      res.status(201).json(program);
    } catch (error) {
      res.status(500).json({ error: "Failed to create program" });
    }
  });

  app.patch("/api/programs/:id", async (req, res) => {
    try {
      const program = await storage.updateProgram(req.params.id, req.body);
      if (!program) {
        return res.status(404).json({ error: "Program not found" });
      }
      res.json(program);
    } catch (error) {
      res.status(500).json({ error: "Failed to update program" });
    }
  });

  app.delete("/api/programs/:id", async (req, res) => {
    try {
      const success = await storage.deleteProgram(req.params.id);
      if (!success) {
        return res.status(404).json({ error: "Program not found" });
      }
      res.json({ success: true });
    } catch (error) {
      res.status(500).json({ error: "Failed to delete program" });
    }
  });

  app.post("/api/programs/:id/evidences", async (req, res) => {
    try {
      const program = await storage.getProgram(req.params.id);
      if (!program) {
        return res.status(404).json({ error: "Program not found" });
      }
      
      const { objectPaths } = req.body;
      if (!objectPaths || !Array.isArray(objectPaths)) {
        return res.status(400).json({ error: "objectPaths array is required" });
      }
      
      const existingEvidences = program.evidences || [];
      const allEvidences = [...existingEvidences, ...objectPaths];
      
      const updated = await storage.updateProgram(req.params.id, { evidences: allEvidences });
      res.json(updated);
    } catch (error) {
      res.status(500).json({ error: "Failed to save evidences" });
    }
  });

  app.delete("/api/programs/:id/evidences/:index", async (req, res) => {
    try {
      const program = await storage.getProgram(req.params.id);
      if (!program) {
        return res.status(404).json({ error: "Program not found" });
      }
      
      const index = parseInt(req.params.index);
      const evidences = program.evidences || [];
      
      if (index < 0 || index >= evidences.length) {
        return res.status(400).json({ error: "Invalid evidence index" });
      }
      
      const newEvidences = evidences.filter((_, i) => i !== index);
      const updated = await storage.updateProgram(req.params.id, { 
        evidences: newEvidences.length > 0 ? newEvidences : null 
      });
      res.json(updated);
    } catch (error) {
      res.status(500).json({ error: "Failed to delete evidence" });
    }
  });

  app.get("/api/loans", async (req, res) => {
    try {
      const loans = await storage.getLoans();
      res.json(loans);
    } catch (error) {
      res.status(500).json({ error: "Failed to fetch loans" });
    }
  });

  app.post("/api/loans", async (req, res) => {
    try {
      const loan = await storage.createLoan(req.body);
      res.status(201).json(loan);
    } catch (error) {
      res.status(500).json({ error: "Failed to create loan" });
    }
  });

  app.patch("/api/loans/:id", async (req, res) => {
    try {
      const loan = await storage.updateLoan(req.params.id, req.body);
      if (!loan) {
        return res.status(404).json({ error: "Loan not found" });
      }
      res.json(loan);
    } catch (error) {
      res.status(500).json({ error: "Failed to update loan" });
    }
  });

  app.delete("/api/loans/:id", async (req, res) => {
    try {
      const deleted = await storage.deleteLoan(req.params.id);
      if (!deleted) {
        return res.status(404).json({ error: "Loan not found" });
      }
      res.json({ success: true });
    } catch (error) {
      res.status(500).json({ error: "Failed to delete loan" });
    }
  });

  app.get("/api/visits", async (req, res) => {
    try {
      const semester = req.query.semester ? parseInt(req.query.semester as string) : undefined;
      const visits = await storage.getClassroomVisits(semester);
      res.json(visits);
    } catch (error) {
      res.status(500).json({ error: "Failed to fetch visits" });
    }
  });

  app.post("/api/visits", async (req, res) => {
    try {
      const visit = await storage.createClassroomVisit(req.body);
      res.status(201).json(visit);
    } catch (error) {
      res.status(500).json({ error: "Failed to create visit" });
    }
  });

  app.patch("/api/visits/:id", async (req, res) => {
    try {
      const visit = await storage.updateClassroomVisit(req.params.id, req.body);
      if (!visit) {
        return res.status(404).json({ error: "Visit not found" });
      }
      res.json(visit);
    } catch (error) {
      res.status(500).json({ error: "Failed to update visit" });
    }
  });

  app.delete("/api/visits/:id", async (req, res) => {
    try {
      const deleted = await storage.deleteClassroomVisit(req.params.id);
      if (!deleted) {
        return res.status(404).json({ error: "Visit not found" });
      }
      res.json({ success: true });
    } catch (error) {
      res.status(500).json({ error: "Failed to delete visit" });
    }
  });

  app.get("/api/reading", async (req, res) => {
    try {
      const semester = req.query.semester ? parseInt(req.query.semester as string) : undefined;
      const sessions = await storage.getReadingSessions(semester);
      res.json(sessions);
    } catch (error) {
      res.status(500).json({ error: "Failed to fetch reading sessions" });
    }
  });

  app.post("/api/reading", async (req, res) => {
    try {
      const session = await storage.createReadingSession(req.body);
      res.status(201).json(session);
    } catch (error) {
      res.status(500).json({ error: "Failed to create reading session" });
    }
  });

  app.patch("/api/reading/:id", async (req, res) => {
    try {
      const session = await storage.updateReadingSession(req.params.id, req.body);
      if (!session) {
        return res.status(404).json({ error: "Reading session not found" });
      }
      res.json(session);
    } catch (error) {
      res.status(500).json({ error: "Failed to update reading session" });
    }
  });

  app.delete("/api/reading/:id", async (req, res) => {
    try {
      const deleted = await storage.deleteReadingSession(req.params.id);
      if (!deleted) {
        return res.status(404).json({ error: "Reading session not found" });
      }
      res.json({ success: true });
    } catch (error) {
      res.status(500).json({ error: "Failed to delete reading session" });
    }
  });

  app.get("/api/books", async (req, res) => {
    try {
      const books = await storage.getBooks();
      res.json(books);
    } catch (error) {
      res.status(500).json({ error: "Failed to fetch books" });
    }
  });

  app.post("/api/books", async (req, res) => {
    try {
      const book = await storage.createBook(req.body);
      res.status(201).json(book);
    } catch (error) {
      res.status(500).json({ error: "Failed to create book" });
    }
  });

  app.patch("/api/books/:id", async (req, res) => {
    try {
      const book = await storage.updateBook(req.params.id, req.body);
      if (!book) {
        return res.status(404).json({ error: "Book not found" });
      }
      res.json(book);
    } catch (error) {
      res.status(500).json({ error: "Failed to update book" });
    }
  });

  app.delete("/api/books/:id", async (req, res) => {
    try {
      const deleted = await storage.deleteBook(req.params.id);
      if (!deleted) {
        return res.status(404).json({ error: "Book not found" });
      }
      res.json({ success: true });
    } catch (error) {
      res.status(500).json({ error: "Failed to delete book" });
    }
  });

  app.get("/api/dfd/english.pdf", async (req, res) => {
    try {
      const pdfBuffer = await generateDFDPdfEnglish();
      res.setHeader("Content-Type", "application/pdf");
      res.setHeader("Content-Disposition", "attachment; filename=DFD_English.pdf");
      res.send(pdfBuffer);
    } catch (error) {
      console.error("Error generating English PDF:", error);
      res.status(500).json({ error: "Failed to generate PDF" });
    }
  });

  app.get("/api/dfd/arabic.pdf", async (req, res) => {
    try {
      const pdfBuffer = await generateDFDPdfArabic();
      res.setHeader("Content-Type", "application/pdf");
      res.setHeader("Content-Disposition", "attachment; filename=DFD_Arabic.pdf");
      res.send(pdfBuffer);
    } catch (error) {
      console.error("Error generating Arabic PDF:", error);
      res.status(500).json({ error: "Failed to generate PDF" });
    }
  });

  return httpServer;
}
